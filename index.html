export default { async fetch(request, env, ctx) { const url = new URL(request.url); url.hostname = 'www.google.com'; const newHeaders = new Headers(request.headers); newHeaders.set('Host', 'www.google.com'); newHeaders.set('Referer', 'https://www.google.com/'); const newRequest = new Request(url, { method: request.method, headers: newHeaders, body: request.body, redirect: 'follow' }); const response = await fetch(newRequest); const contentType = response.headers.get('content-type') || ''; if (contentType.includes('text/html')) { let body = await response.text(); const myHost = new URL(request.url).host; body = body.replace(/www\.google\.com/g, myHost); return new Response(body, { headers: response.headers }); } return response; }, };
